359 static inline BOOLEAN hdmi_info_IsValid(HDMI_INFOFRAME_PACKET *info)
360 {
361  unsigned char *pbyData = (unsigned char *)&info->header;
362  BYTE cbData = info->header.byLength + sizeof(info->header) + 1;
363 
364  BYTE bySum = 0;
365  while (cbData-- != 0)
366  bySum += *pbyData++;
367 
368  return (bySum == 0);
369 }
